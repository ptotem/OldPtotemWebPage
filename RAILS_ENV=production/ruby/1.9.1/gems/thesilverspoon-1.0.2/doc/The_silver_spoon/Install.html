def create_images

  # copies the standard images into the assets/images folder
  @images=Array.new
  @images= Dir.entries("#{Install.source_root}/assets/images")
  @images.each do |image|

    if image!=".." and image !="."
      copy_file "assets/images/#{image.to_s}", "app/assets/images/#{image}"
    end
  end


end
